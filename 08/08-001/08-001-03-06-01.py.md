# `base`进制`m` -> 十进制`n`

```python
from random import randint

base = 2
length = 8
print(f"{base = }, {length = }")
m = ""
for i in range(length):
    m += str(randint(0, base - 1))
print(f"{m = }")
print(f"n  = {int(m, base=base)}")

n1 = 0
for i in range(length):
    n1 += int(m[i]) * base ** (length - (i + 1))
print(f"{n1 = }")

n2 = 0
for i in range(length):
    n2 = int(m[i]) + n2 * base
print(f"{n2 = }")
```
