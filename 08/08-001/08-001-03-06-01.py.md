# `base`进制`m` -> 十进制`n`

```python
from random import randint

base = 2
length = 8
print(f"{base = }, {length = }")
m_lst = [randint(0, base - 1) for i in range(length)]
m = "".join([str(i) for i in m_lst])
print(f"{m = }")
print(f"n  = {int(m, base=base)}")

n1 = 0
for i in range(length):
    n1 += m_lst[i] * base ** (length - (i + 1))
print(f"{n1 = }")

n2 = 0
for i in range(length):
    n2 = m_lst[i] + n2 * base
print(f"{n2 = }")
```
