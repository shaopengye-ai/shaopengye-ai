# 十进制`n` -> `base`进制`m`

```python
from random import randint

n = randint(0, 1000)
print(f"{n = }")

base = 2
print(f"m = '{bin(n)}'")

stack = []
while True:
    q = n // base
    r = n - q * base
    stack.append(str(r))
    if q == 0:
        break
    
    n = q
m = "0b"
for i in range(len(stack)):
    m += stack.pop()
print(f"{m = }")
```
